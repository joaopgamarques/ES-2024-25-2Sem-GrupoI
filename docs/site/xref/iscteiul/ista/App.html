115     /**
116      * Main method: loads the CSV, filters by parish, builds a Graph,
117      * picks a random property from the subset, and compares adjacency using two methods.
118      *
119      * @param args command-line arguments (not used)
120      */
121     public static void main(String[] args) {
122         // 1. Read proprieties CSV
123         CSVFileReader csvFileReader = new CSVFileReader();
124         propertyRecords = csvFileReader.importData("/Madeira-Moodle-1.2.csv");
125         logger.info("Total records loaded: {}", propertyRecords.size());
126         propertyRecords.removeIf(pr -> pr.getParish() == null || pr.getParish().isBlank());
127 
128         // 1a. Print distinct parishes and municipalities.
129         Set<String> distinctParishes = PropertyUtils.getDistinctParishes(propertyRecords);
130         Set<String> distinctMunicipalities = PropertyUtils.getDistinctMunicipalities(propertyRecords);
131         logger.info("Distinct Parishes: {}", distinctParishes);
132         logger.info("Distinct Municipalities: {}", distinctMunicipalities);
133 
134         // 1b) Find objectID=11074 in the data, copy it into our private static field
135         funchalPropertyRecord = propertyRecords.stream()
136                 .filter(pr -> pr.getObjectID() == 11074)
137                 .findFirst()
138                 .map(original -> new PropertyRecord(original.getObjectID(), original.getParcelID(),
139                         original.getParcelNumber(), original.getShapeLength(), original.getShapeArea(), original.getGeometry(),
140                         original.getOwner(), original.getParish(), original.getMunicipality(), original.getIsland()))
141                 .orElse(null);
142 
143         // 1c) Find objectID=11517 in the data, copy it into our private static field
144         machicoPropertyRecord = propertyRecords.stream()
145                 .filter(pr -> pr.getObjectID() == 11517)
146                 .findFirst()
147                 .map(original -> new PropertyRecord(original.getObjectID(), original.getParcelID(),
148                         original.getParcelNumber(), original.getShapeLength(), original.getShapeArea(), original.getGeometry(),
149                         original.getOwner(), original.getParish(), original.getMunicipality(), original.getIsland()))
150                 .orElse(null);
151 
152         // 1d. Read metrics CSV
153         List<ParishMetrics> metrics = CSVMetricsFileReader.importData();
154         logger.info("Total metrics loaded: {}", metrics.size());
155 
156         // 1c. Print each parish metric (distance, price, etc.)
157         for (ParishMetrics pm : metrics) {
158             System.out.printf("%-30s Distance Airport [km]=%.1f | Distance Funchal Sé [km]=%.1f | Price [€/m²]=%.0f | " +
159                             "Population Density [Hab./km²]=%d | Infrastructure Quality Index=%d%n",
160                     pm.parishName(),
161                     pm.distanceAirportKm(),
162                     pm.distanceFunchalSeKm(),
163                     pm.averagePriceEuroM2(),
164                     pm.populationDensityHabKm2(),
165                     pm.infrastructureQualityIdx());
166         }
167 
168         // 2. Filter to a chosen parish.
169         String chosenParish = "Machico";
170         List<PropertyRecord> parishSubset = propertyRecords.stream()
171                 .filter(pr -> chosenParish.equals(pr.getParish()))
172                 .collect(Collectors.toList());
173         logger.info("Records in parish '{}': {}", chosenParish, parishSubset.size());
174 
175         // 2a. Calculate the average area of properties in the chosen parish (no adjacency grouping).
176         List<PropertyRecord> parishProperties = PropertyUtils.findByParish(propertyRecords, chosenParish);
177         double averageArea = PropertyUtils.calculateAverageArea(parishProperties);
178         logger.info("Records in parish '{}': {}", chosenParish, parishProperties.size());
179         logger.info("Average area (no adjacency grouping) in '{}' [ha]: {}", chosenParish, String.format("%.2f", averageArea));
180 
181         // 3. Build the custom (O(N²)) Graph from the parish subset.
182         Graph propertyGraph = new Graph(parishSubset);
183 
184         // 4. Pick a random PropertyRecord from the subset.
185         if (parishSubset.isEmpty()) {
186             logger.warn("No records found in parish '{}'. Exiting.", chosenParish);
187             return;
188         }
189 
190         // 4a. For each property in parishSubset, print its neighbors
191         for (PropertyRecord record : parishSubset) {
192             int objectID = record.getObjectID();
193             List<Graph.GraphNode> neighbors = propertyGraph.getNeighbors(objectID);
194             if (neighbors.isEmpty()) {
195                 System.out.println("Property ID=" + objectID
196                                 + ", Area [ha]=" + String.format("%.2f", record.getShapeArea())
197                                 + ", Owner=" + record.getOwner() + " has no neighbors.");
198             } else {
199                 System.out.println("Neighbors of Property ID=" + objectID
200                         + ", Area [ha]=" + String.format("%.2f", record.getShapeArea()) + ", Owner=" + record.getOwner() + ":");
201                 for (Graph.GraphNode neighbor : neighbors) {
202                     System.out.println("  -> objectID=" + neighbor.getObjectID()
203                             + ", Area [ha]=" + String.format("%.2f", neighbor.getShapeArea())
204                             + ", Owner=" + neighbor.getOwner());
205                 }
206             }
207         }
208 
209         Random random = new Random();
210         int randomIndex = random.nextInt(parishSubset.size());
211         PropertyRecord randomProperty = parishSubset.get(randomIndex);
212         int testObjectID = randomProperty.getObjectID();
213 
214         logger.info("Randomly chosen objectID {} from parish '{}'", testObjectID, chosenParish);
215 
216         // 4a. Parse geometry and print centroid + other attributes for just this node.
217         WKTReader wktReader = new WKTReader();
218         String wkt = randomProperty.getGeometry();
219         if (wkt != null) {
220             try {
221                 Geometry geom = wktReader.read(wkt);
222                 Point centroid = geom.getCentroid();
223                 double cx = centroid.getX();
224                 double cy = centroid.getY();
225 
226                 logger.info("Selected node attributes:");
227                 logger.info(" -> Object ID      = {}", randomProperty.getObjectID());
228                 logger.info(" -> Parcel ID      = {}", randomProperty.getParcelID());
229                 logger.info(" -> Perimeter [km] = {}", randomProperty.getShapeLength());
230                 logger.info(" -> Area [ha]      = {}", randomProperty.getShapeArea());
231                 logger.info(" -> Owner          = {}", randomProperty.getOwner());
232                 logger.info(" -> Parish         = {}", randomProperty.getParish());
233                 logger.info(" -> Municipality   = {}", randomProperty.getMunicipality());
234                 logger.info(" -> Island         = {}", randomProperty.getIsland());
235                 logger.info(" -> Centroid       = ({}, {})", cx, cy);
236 
237             } catch (ParseException e) {
238                 logger.warn("Failed to parse WKT for objectID={}: {}",
239                         randomProperty.getObjectID(), e.getMessage());
240             }
241         }
242 
243         // 5. Get adjacency from the Graph.
244         List<Graph.GraphNode> graphNeighbors = propertyGraph.getNeighbors(testObjectID);
245         Set<Integer> graphNeighborIDs = graphNeighbors.stream()
246                 .map(Graph.GraphNode::getObjectID)
247                 .collect(Collectors.toSet());
248 
249         // 6. List-based adjacency check for the same record (using parishSubset).
250         List<PropertyRecord> listNeighbors = PropertyUtils.findAdjacentProperties(randomProperty, parishSubset);
251         Set<Integer> listNeighborIDs = listNeighbors.stream()
252                 .map(PropertyRecord::getObjectID)
253                 .collect(Collectors.toSet());
254 
255         // 7. Compare the sets.
256         if (graphNeighborIDs.equals(listNeighborIDs)) {
257             logger.info("Both methods agree on adjacency for objectID = {}", testObjectID);
258         } else {
259             logger.warn("Mismatch in adjacency sets for objectID {}:\nGraph-based: {}\nList-based: {}",
260                     testObjectID, graphNeighborIDs, listNeighborIDs);
261         }
262 
263         // 8. Build the JGraphT-based PropertyGraph (with STRtree-based adjacency).
264         PropertyGraph propertyGraphJgt = new PropertyGraph();
265         propertyGraphJgt.buildGraph(parishSubset);
266 
267         // 8a. Calculate the average area of properties grouped by owner.
268         org.jgrapht.Graph<PropertyRecord, DefaultEdge> jgtGraph = propertyGraphJgt.getGraph();
269         double averageGroupedArea = PropertyUtils.calculateAverageGroupedArea(parishSubset, jgtGraph);
270         System.out.println("Average area of properties in parish (grouped by owner) [ha]: " + String.format("%.2f",averageGroupedArea));
271         if (averageArea != averageGroupedArea) {
272             System.out.println("These averages differ: " + String.format("%.2f",averageArea) + " ha" + " vs "
273                     + String.format("%.2f",averageGroupedArea) + " ha");
274         }
275 
276         // 8b. Demonstrate merging adjacent properties that belong to the same owner, within this parish subset.
277         demonstrateMergingProperties(parishSubset);
278 
279         // 8c) Merge same-owner adjacency
280         List<PropertyRecord> merged = PropertyMerger.mergeSameOwner(parishSubset);
281 
282         // 8d) Build adjacency among these merged props
283         SimpleGraph<PropertyRecord, DefaultEdge> mergedGraph = MergedPropertyGraph.buildGraph(merged);
284 
285         // 8e) Suggest swaps with areaThreshold=0.1 => up to 10 suggestions
286         List<SwapSuggestion> suggestions = PropertySwapAdvisor.suggestSwaps(mergedGraph, 0.1, 10);
287 
288         // 8f) Print the suggestions
289         for (SwapSuggestion s : suggestions) {
290             System.out.println(s);
291         }
292 
293         // 9. Build the owner graph using the same parish subset.
294         OwnerGraph ownerGraph = new OwnerGraph();
295         ownerGraph.buildGraph(parishSubset);
296 
297         // 10. Check how many owners are in the graph.
298         System.out.println("Number of owners in the OwnerGraph: " + ownerGraph.getOwners().size());
299 
300         // 11. Pick one owner's ID (from the first property) and see who they're adjacent to.
301         if (!parishSubset.isEmpty()) {
302             int someOwner = parishSubset.get(0).getOwner();
303             Set<Integer> ownerNeighbors = ownerGraph.getNeighbors(someOwner);
304             System.out.println("Owner " + someOwner + " is adjacent to owners: " + ownerNeighbors);
305         }
306 
307         // 12. Calculate the distance to Funchal Sé.
308         double distanceToFunchal = PropertyUtils.distanceToFunchal(1234);
309         System.out.println("Distance to Funchal Sé in kilometers: " + String.format("%.1f", distanceToFunchal/1000));
310 
311         // 12a. Calculate the distance to Machico.
312         double distanceMachico = PropertyUtils.distanceToMachico(1234);
313         System.out.println("Distance to Machico in kilometers: " + String.format("%.1f", distanceMachico/1000));
314 
315         // 13. Visualize the STRtree-based property graph in GraphStream.
316         // GraphVisualization.visualizeGraph(propertyGraphJgt);
317 
318         // 14. Create a swing-based UI for the application
319         AppUI.showUI();
320 
321     }

323     /**
324      * Demonstrates merging adjacent properties belonging to the same owner.
325      * <p>
326      * This example filters properties by a chosen parish, then merges them
327      * to produce a new list where each connected component is collapsed
328      * into a single {@link PropertyRecord}.
329      */
330     private static void demonstrateMergingProperties(List<PropertyRecord> parishSubset) {
331         System.out.println("\n--- Demonstrate Merging of Adjacent Properties ---");
332         List<PropertyRecord> merged = PropertyUtils.mergeAdjacentPropertiesSameOwner(parishSubset);
333 
334         System.out.println("Original subset size: " + parishSubset.size());
335         System.out.println("Merged subset size: " + merged.size());
336         for (PropertyRecord mp : merged) {
337             System.out.println("Merged property => Object ID=" + mp.getObjectID()
338                     + ", Area [ha]=" + mp.getShapeArea()
339                     + ", Geometry=" + mp.getGeometry().substring(0, Math.min(60, mp.getGeometry().length())) + "...");
340         }
